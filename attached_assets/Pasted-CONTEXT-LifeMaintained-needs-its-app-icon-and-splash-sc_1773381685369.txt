javascriptconst sharp = require('sharp');
const path = require('path');

const LOGO = path.join(__dirname, '..', 'assets', 'images', 'brand-logo.png');
const OUT = path.join(__dirname, '..', 'assets', 'images');

async function generate() {
  // 1. App icon: logo centered on white background, 1024x1024
  await sharp(LOGO)
    .resize(680, 680, { fit: 'inside' })
    .flatten({ background: '#FFFFFF' })
    .extend({
      top: 172, bottom: 172, left: 172, right: 172,
      background: '#FFFFFF'
    })
    .resize(1024, 1024)
    .png()
    .toFile(path.join(OUT, 'icon.png'));

  // 2. Android foreground: logo centered on transparent, 512x512
  // Logo in center 66% (338px) with padding
  await sharp(LOGO)
    .resize(338, 338, { fit: 'inside' })
    .extend({
      top: 87, bottom: 87, left: 87, right: 87,
      background: { r: 0, g: 0, b: 0, alpha: 0 }
    })
    .resize(512, 512)
    .png()
    .toFile(path.join(OUT, 'android-icon-foreground.png'));

  // 3. Android background: solid white, 512x512
  await sharp({
    create: { width: 512, height: 512, channels: 4, background: '#FFFFFF' }
  }).png().toFile(path.join(OUT, 'android-icon-background.png'));

  // 4. Android monochrome: white silhouette, 512x512
  await sharp(LOGO)
    .resize(338, 338, { fit: 'inside' })
    .threshold(128)
    .negate()
    .extend({
      top: 87, bottom: 87, left: 87, right: 87,
      background: { r: 0, g: 0, b: 0, alpha: 0 }
    })
    .resize(512, 512)
    .png()
    .toFile(path.join(OUT, 'android-icon-monochrome.png'));

  // 5. Favicon: logo on white, 48x48
  await sharp(LOGO)
    .resize(32, 32, { fit: 'inside' })
    .flatten({ background: '#FFFFFF' })
    .extend({
      top: 8, bottom: 8, left: 8, right: 8,
      background: '#FFFFFF'
    })
    .resize(48, 48)
    .png()
    .toFile(path.join(OUT, 'favicon.png'));

  // 6. Splash icon: logo on transparent, 512x512
  await sharp(LOGO)
    .resize(400, 400, { fit: 'inside' })
    .extend({
      top: 56, bottom: 56, left: 56, right: 56,
      background: { r: 0, g: 0, b: 0, alpha: 0 }
    })
    .resize(512, 512)
    .png()
    .toFile(path.join(OUT, 'splash-icon.png'));

  console.log('All icons generated successfully');
}

generate().catch(console.error);
Adjust the resize values if the output doesn't look right — the logo should be centered with comfortable padding and never touch the edges. Run the script: node scripts/generate-icons.js
