typescript// Tier hierarchy — higher index = more access
const TIER_HIERARCHY = ['free', 'personal', 'pro', 'business'];

// Returns true if user has any active paid access (trial OR any paid tier)
export function hasActivePremium(profile: Profile): boolean {
  if (profile.subscription_tier === 'trial' && new Date(profile.trial_expires_at) > new Date()) return true;
  if (['personal', 'pro', 'business'].includes(profile.subscription_tier) && 
      profile.subscription_expires_at && new Date(profile.subscription_expires_at) > new Date()) return true;
  return false;
}

// Returns true if user has at least Personal tier active
export function hasPersonalOrAbove(profile: Profile): boolean {
  return hasActivePremium(profile);
}

// Returns true if user has Pro tier or above
export function hasProOrAbove(profile: Profile): boolean {
  if (profile.subscription_tier === 'trial' && new Date(profile.trial_expires_at) > new Date()) return true;
  if (['pro', 'business'].includes(profile.subscription_tier) && 
      profile.subscription_expires_at && new Date(profile.subscription_expires_at) > new Date()) return true;
  return false;
}

// Returns true if user has Business tier
export function hasBusiness(profile: Profile): boolean {
  if (profile.subscription_tier === 'trial' && new Date(profile.trial_expires_at) > new Date()) return true;
  if (profile.subscription_tier === 'business' && 
      profile.subscription_expires_at && new Date(profile.subscription_expires_at) > new Date()) return true;
  return false;
}

// Vehicle limit by tier
export function vehicleLimit(profile: Profile): number {
  if (hasBusiness(profile)) return Infinity;
  if (hasProOrAbove(profile)) return 6;
  if (hasPersonalOrAbove(profile)) return 3;
  return 1;
}

// Property limit by tier
export function propertyLimit(profile: Profile): number {
  if (hasBusiness(profile)) return Infinity;
  if (hasProOrAbove(profile)) return 5;
  if (hasPersonalOrAbove(profile)) return 2;
  return 1;
}

// Monthly scan limit by tier
export function scanLimit(profile: Profile): number {
  if (hasBusiness(profile)) return 100;
  if (hasProOrAbove(profile)) return 30;
  if (hasPersonalOrAbove(profile)) return 15;
  return 0;
}

export function scansRemaining(profile: Profile): number {
  return Math.max(0, scanLimit(profile) - (profile.monthly_scan_count ?? 0));
}
