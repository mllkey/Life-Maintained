function handleScanComplete(result: ReceiptScanResult) {
  if (result.date) setDate(result.date);
  if (result.mileage != null) setMileage(String(result.mileage));
  if (result.provider) setProvider(result.provider);
  
  if (result.items && result.items.length > 1) {
    // Multiple services found - store them for batch insert
    setScannedItems(result.items);
    setCost(result.cost != null ? String(result.cost) : "");
    setTask(result.task || "");
  } else {
    // Single service - fill form normally
    if (result.task) setTask(result.task);
    if (result.cost != null) setCost(String(result.cost));
    setScannedItems([]);
  }
  setOcrApplied(true);
  Haptics.notificationAsync(Haptics.NotificationFeedbackType.Success);
}

3. Update handleSave: if scannedItems has multiple items, insert a separate maintenance_logs row for each item, all with the same date, provider, mileage, and vehicle_id. Each row gets its own task name and cost from the items array:

async function handleSave() {
  if (!user || !vehicleId) return;
  setIsLoading(true);
  setError(null);

  try {
    if (scannedItems.length > 1) {
      // Batch insert - one row per service item
      const rows = scannedItems.map(item => ({
        vehicle_id: vehicleId,
        task: item.name,
        date: date || new Date().toISOString().split("T")[0],
        mileage: mileage ? parseInt(mileage) : null,
        cost: item.cost,
        provider: provider.trim() || null,
        notes: null,
        created_at: new Date().toISOString(),
        updated_at: new Date().toISOString(),
      }));
      
      const { error: err } = await supabase.from("maintenance_logs").insert(rows);
      if (err) throw err;
    } else {
      // Single entry
      if (!task.trim()) {
        setError("Service description is required");
        setIsLoading(false);
        return;
      }
      const { error: err } = await supabase.from("maintenance_logs").insert({
        vehicle_id: vehicleId,
        task: task.trim(),
        date: date || new Date().toISOString().split("T")[0],
        mileage: mileage ? parseInt(mileage) : null,
        cost: cost ? parseFloat(cost) : null,
        provider: provider.trim() || null,
        notes: notes.trim() || null,
        created_at: new Date().toISOString(),
        updated_at: new Date().toISOString(),
      });
      if (err) throw err;
    }

    // Update vehicle mileage if provided
    if (mileage) {
      await supabase.from("vehicles").update({
        mileage: parseInt(mileage),
        updated_at: new Date().toISOString(),
      }).eq("id", vehicleId);
      await supabase.from("vehicle_mileage_history").insert({
        vehicle_id: vehicleId,
        mileage: parseInt(mileage),
        recorded_at: date || new Date().toISOString(),
        created_at: new Date().toISOString(),
      });
    }

    Haptics.notificationAsync(Haptics.NotificationFeedbackType.Success);
    queryClient.invalidateQueries({ queryKey: ["maintenance_logs", vehicleId] });
    queryClient.invalidateQueries({ queryKey: ["vehicle", vehicleId] });
    queryClient.invalidateQueries({ queryKey: ["vehicles"] });
    queryClient.invalidateQueries({ queryKey: ["dashboard"] });
    router.back();
  } catch (err: any) {
    setError(err.message);
    Haptics.notificationAsync(Haptics.NotificationFeedbackType.Error);
  } finally {
    setIsLoading(false);
  }
}
